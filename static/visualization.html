     LARGE SCHEMA DETECTION & REPORT MODE

  const LARGE_THRESHOLD = 150;
  const HUGE_THRESHOLD = 400;
  let reportAvailable = false;
  let esriReportFile = null;

  async function probeReportAssets() {
    const candidates = [
      'esri_schema_report.html',
      'schema_report.html',
      'SchemaReport.html'
    ];
    for (const name of candidates) {
      try {
        const resp = await fetch(name, { method: 'HEAD' });
        if (resp.ok) { esriReportFile = name; reportAvailable = true; return; }
      } catch (e) { /* ignore */ }
    }
  }

  function checkSchemaSize(nodeCount) {
    const banner = document.getElementById('large-schema-banner');
    if (nodeCount < LARGE_THRESHOLD) { banner.style.display = 'none'; return; }

    const icon = banner.querySelector('.banner-icon');
    const text = banner.querySelector('.banner-text');
    const actions = banner.querySelector('.banner-actions');
    actions.innerHTML = '';

    if (nodeCount >= HUGE_THRESHOLD) {
      banner.className = 'critical';
      icon.textContent = '\u26A0\uFE0F';
      text.innerHTML = 'Large Schema (' + nodeCount + ' nodes) \u2014 Interactive diagrams may be slow or cluttered.';

      if (reportAvailable) {
        actions.innerHTML =
          '' +
          '';
      } else {
        text.innerHTML += ' Generate an Esri Schema Report for best results:
gdbviz extract -p <gdb> --schema-report';
      }
    } else {
      banner.className = 'warn';
      icon.textContent = '\u26A0\uFE0F';
      text.innerHTML = 'Moderate Schema (' + nodeCount + ' nodes) \u2014 Diagrams may be cluttered.';
      if (reportAvailable) {
        actions.innerHTML =
          '';
      }
    }

    updateBannerRightEdge();
  }

  function updateBannerRightEdge() {
    const banner = document.getElementById('large-schema-banner');
    const panel = document.getElementById('detail-panel');
    if (!banner || banner.style.display === 'none') return;
    if (panel && !panel.classList.contains('collapsed')) {
      banner.style.right = '340px';
    } else {
      banner.style.right = '0';
    }
  }

  function showReportMode() {
    document.getElementById('canvas-wrap').style.display = 'none';
    document.getElementById('sub-header').style.display = 'none';
    document.getElementById('detail-panel').style.display = 'none';
    document.getElementById('legend').style.display = 'none';
    document.getElementById('reset-btn').style.display = 'none';
    document.querySelectorAll('.format-view').forEach(v => v.style.display = 'none');

    const rm = document.getElementById('report-mode');
    rm.classList.add('active');

    const body = document.getElementById('report-body');
    if (esriReportFile) {
      body.innerHTML = '';
    } else {
      body.innerHTML = '
No Esri Schema Report found.
Generate one with: gdbviz extract -p <gdb> --schema-report
';
    }

    document.querySelectorAll('.fmt-tab').forEach(t => t.classList.remove('active'));
  }

  function hideReportMode() {
    document.getElementById('report-mode').classList.remove('active');
    document.getElementById('report-body').innerHTML = '';
    switchFormat('d3');
  }

  probeReportAssets().then(() => {
    if (SCHEMA && SCHEMA.nodes) checkSchemaSize(SCHEMA.nodes.length);
  });
